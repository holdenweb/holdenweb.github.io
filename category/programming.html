from PIL import Image, ImageDraw
UNIT_WIDTH = 12  # Pixels per story point
UNIT_HEIGHT = 20 # Height of bar in pixels
blobs = []
def issue_bar(points):
    pos = 0
    tot_len = 0
    for typ, issues in points.items():
        blobs = [1 if i is None else int(i) for i in issues]
        tot_len += int(sum(b for b in blobs))
    img = Image.new('RGB', (tot_len*UNIT_WIDTH, UNIT_HEIGHT))
    #print("Image size", (tot_len*UNIT_WIDTH, UNIT_HEIGHT))
    for typ in 'TICQDW':
        try:
            issues = points[typ]
            blobs = [1 if i is None else int(i) for i in issues]
            draw = ImageDraw.ImageDraw(img)
            for b in blobs:
                b *= UNIT_WIDTH
                #print(f"Filling {(pos, 0), (pos+b, UNIT_HEIGHT)} with {colours[typ]}")
                draw.rectangle(((pos, 0), (pos+b, UNIT_HEIGHT)), fill=colours[typ])
                draw.line(((pos+b, 0), (pos+b, UNIT_HEIGHT)), fill='black', width=5)
                pos += b
        except KeyError:
            continue
    return img
